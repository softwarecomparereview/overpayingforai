PART 1 — CREATE AFFILIATE RESOLVER
-----------------------------------

Create file:
/src/utils/affiliateResolver.ts

Requirements:

- Input: tool name (string)
- Output: affiliate link (string)

Logic:
- Load from static config (for now)
- Fallback to "#"

Example:

export const AFFILIATE_MAP: Record<string, string> = {
  "gpt-4o": "https://platform.openai.com/signup",
  "claude": "https://www.anthropic.com/partners",
  "perplexity": "https://www.perplexity.ai",
  "cursor": "https://cursor.sh",
  "notion": "https://www.notion.so/affiliates"
};

export function getAffiliateLink(tool: string): string {
  const key = tool.toLowerCase();
  return AFFILIATE_MAP[key] || "#";
}
